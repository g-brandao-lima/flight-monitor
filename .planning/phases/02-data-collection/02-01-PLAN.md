       - route_group_id: Mapped[int] ForeignKey("route_groups.id")
       - origin: Mapped[str] String(3) -- IATA code

       - departure_date: Mapped[date] Date
       - return_date: Mapped[date] Date

       - currency: Mapped[str] String(3) -- default "BRL"

       - price_third_quartile: Mapped[float | None] Float nullable -- THIRD quartile
       - price_max: Mapped[float | None] Float nullable -- MAXIMUM quartile
       - price_classification: Mapped[str | None] String(10) nullable -- LOW/MEDIUM/HIGH
       - collected_at: Mapped[datetime] DateTime server_default=func.now()
       - route_group: relationship("RouteGroup", backref="snapshots")
       - booking_classes: relationship("BookingClassSnapshot", backref="flight_snapshot", cascade="all, delete-orphan")

       - __tablename__ = "booking_class_snapshots"
       - id: Mapped[int] PK autoincrement
       - flight_snapshot_id: Mapped[int] ForeignKey("flight_snapshots.id")
       - class_code: Mapped[str] String(1) -- Y, B, M, H, Q, K, L

       - segment_direction: Mapped[str] String(10) -- OUTBOUND ou INBOUND
