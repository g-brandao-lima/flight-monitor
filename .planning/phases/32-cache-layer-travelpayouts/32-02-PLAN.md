<travelpayouts_api_reference>
**Token:** header `X-Access-Token: <token>` (ou query ?token=). Usar header.
**Base URL:** https://api.travelpayouts.com

**Endpoint 1 — preco mais barato por destino:**
`GET /v1/prices/cheap?origin=GRU&destination=LIS&currency=brl`
Resposta: `{"success": true, "data": {"LIS": {"1": {"price": 2800, "airline": "TP", "departure_at": "2026-09-01T10:00:00", "return_at": "2026-09-15T14:00:00", ...}}}}`

**Endpoint 2 — preco dia-a-dia do mes:**
`GET /v1/prices/calendar?origin=GRU&destination=LIS&depart_date=2026-09&currency=brl`
Resposta: `{"success": true, "data": {"2026-09-01": {"price": 2800, "airline": "TP", "departure_at": "...", "return_at": "..."}, "2026-09-02": {...}}}`

**Endpoint 3 — matriz do mes:**
`GET /v2/prices/month-matrix?origin=GRU&destination=LIS&currency=brl&show_to_affiliates=true`
Resposta: `{"success": true, "data": [{"depart_date": "2026-09-01", "return_date": "2026-09-15", "value": 2800, ...}, ...]}`

**PEGADINHA CRITICA:** A API expande IATA de aeroporto para IATA de cidade nas respostas (GRU → SAO, JFK → NYC, CGH → SAO). Ex: voce pede origin=GRU e a resposta contem origin="SAO" em alguns payloads. Ao normalizar, SEMPRE preservar o origin/destination que o caller pediu como parametro, NUNCA usar o valor que a API devolve dentro de data. Isso porque o route_cache (Plan 01) usa IATA aeroporto como chave (GRU-LIS, nao SAO-LIS).

Rate limit: sem limite documentado. Espacar calls ~200-500ms entre requests.
</travelpayouts_api_reference>

    ```python
    """Cliente para Travelpayouts Data API.

    Autentica via header X-Access-Token. Normaliza respostas num formato uniforme
    preservando o IATA aeroporto pedido (nao IATA cidade retornado pela API).
    """
    import logging
    from typing import Any

    import httpx  # ou requests, conforme disponivel

    from app.config import settings

    logger = logging.getLogger(__name__)

    BASE_URL = "https://api.travelpayouts.com"


    class TravelpayoutsClient:
        def __init__(self, token: str | None = None, timeout: float = 15.0):
            self._token = token if token is not None else settings.travelpayouts_token
            self._timeout = timeout

        @property
        def is_configured(self) -> bool:
            return bool(self._token)

        def _headers(self) -> dict:
            return {"X-Access-Token": self._token, "Accept": "application/json"}

        def _get(self, path: str, params: dict) -> dict | None:
            """GET com header de auth. Retorna None em caso de erro HTTP ou success=false."""
            try:
                with httpx.Client(timeout=self._timeout) as client:
                    resp = client.get(f"{BASE_URL}{path}", params=params, headers=self._headers())
                    resp.raise_for_status()
                    body = resp.json()
            except httpx.HTTPError as e:
                logger.warning("travelpayouts HTTP error on %s: %s", path, e)
                return None
            if not body.get("success"):
                logger.warning("travelpayouts api failure on %s: %s", path, body.get("error"))
                return None
            return body

        def fetch_calendar(
            self, origin: str, destination: str, depart_month: str, currency: str = "BRL"
        ) -> list[dict]:
            """GET /v1/prices/calendar?origin=&destination=&depart_date=YYYY-MM."""
            body = self._get(
                "/v1/prices/calendar",
                {
                    "origin": origin,
                    "destination": destination,
                    "depart_date": depart_month,
                    "currency": currency.lower(),
                },
            )
            if not body:
                return []
            results = []
            for depart_date, entry in (body.get("data") or {}).items():
                if not entry or "price" not in entry:
                    continue
                return_at = entry.get("return_at")
                return_date = return_at[:10] if return_at else None
                results.append({
                    "origin": origin,          # PRESERVA o IATA pedido (pegadinha!)
                    "destination": destination,
                    "departure_date": depart_date,
                    "return_date": return_date,
                    "min_price": float(entry["price"]),
                    "currency": currency.upper(),
                    "airline": entry.get("airline"),
                })
            return results

        def fetch_cheap(
            self, origin: str, destination: str, currency: str = "BRL"
        ) -> list[dict]:
            """GET /v1/prices/cheap?origin=&destination=."""
            body = self._get(
                "/v1/prices/cheap",
                {"origin": origin, "destination": destination, "currency": currency.lower()},
            )
            if not body:
                return []
            results = []
            destinations_block = body.get("data") or {}
            for _dest_code, offers in destinations_block.items():
                for _idx, offer in (offers or {}).items():
                    if "price" not in offer:
                        continue
                    dep_at = offer.get("departure_at")
                    ret_at = offer.get("return_at")
                    results.append({
                        "origin": origin,
                        "destination": destination,
                        "departure_date": dep_at[:10] if dep_at else None,
                        "return_date": ret_at[:10] if ret_at else None,
                        "min_price": float(offer["price"]),
                        "currency": currency.upper(),
                        "airline": offer.get("airline"),
                    })
            return results

        def fetch_month_matrix(
            self, origin: str, destination: str, currency: str = "BRL"
        ) -> list[dict]:
            """GET /v2/prices/month-matrix?origin=&destination=."""
            body = self._get(
                "/v2/prices/month-matrix",
                {
                    "origin": origin,
                    "destination": destination,
                    "currency": currency.lower(),
                    "show_to_affiliates": "true",
                },
            )
            if not body:
                return []
            results = []
            for entry in body.get("data") or []:
                if "value" not in entry and "price" not in entry:
                    continue
                price = entry.get("value") or entry.get("price")
                results.append({
                    "origin": origin,
                    "destination": destination,
                    "departure_date": entry.get("depart_date"),
                    "return_date": entry.get("return_date"),
                    "min_price": float(price),
                    "currency": currency.upper(),
                    "airline": entry.get("airline"),
                })
            return results
    ```
