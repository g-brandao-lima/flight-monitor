    ```python
    """Servico de busca de precos multi-trecho (Phase 36).

    Reusa cache_service, flight_search e price_prediction_service (Phase 34).
    Implementa produto cartesiano limitado de datas por leg (cap 7) com filtro
    por validade temporal (min_stay/max_stay).
    """
    from __future__ import annotations

    import itertools
    import logging
    from datetime import date, timedelta
    from typing import Any

    from sqlalchemy.orm import Session

    from app.models import FlightSnapshot, RouteGroup, RouteGroupLeg
    from app.services import flight_search, route_cache_service, snapshot_service

    logger = logging.getLogger(__name__)

    DATE_CANDIDATES_PER_LEG = 7


    def _candidate_dates(leg: RouteGroupLeg) -> list[date]:
        total_days = (leg.window_end - leg.window_start).days
        if total_days < 0:
            return []
        if total_days <= DATE_CANDIDATES_PER_LEG - 1:
            return [leg.window_start + timedelta(days=i) for i in range(total_days + 1)]
        step = total_days // (DATE_CANDIDATES_PER_LEG - 1)
        return [leg.window_start + timedelta(days=i * step) for i in range(DATE_CANDIDATES_PER_LEG)]


    def _is_valid_chain(dates: tuple[date, ...], legs: list[RouteGroupLeg]) -> bool:
        for i in range(1, len(dates)):
            gap = (dates[i] - dates[i - 1]).days
            prev = legs[i - 1]
            if gap < prev.min_stay_days:
                return False
            if prev.max_stay_days is not None and gap > prev.max_stay_days:
                return False
        return True


    def _fetch_leg_price(db: Session, leg: RouteGroupLeg, dep_date: date, passengers: int) -> dict[str, Any] | None:
        # Cache first (D-12)
        cached = route_cache_service.get_cached_price(
            db,
            origin=leg.origin,
            destination=leg.destination,
            departure_date=dep_date,
        )
        if cached is not None:
            return {"price": float(cached.min_price), "airline": "CACHE", "source": "cache"}

        # Fallback SerpAPI on-demand
        try:
            result = flight_search.search_flights_ex(
                origin=leg.origin,
                destination=leg.destination,
                departure_date=dep_date,
                return_date=dep_date,  # one-way workaround per RESEARCH.md Open Q1
                passengers=passengers,
                max_stops=leg.max_stops,
            )
        except Exception as exc:
            logger.warning("multi_leg leg price fetch failed for leg %s: %s", leg.id, exc)
            return None

        if not result or not getattr(result, "cheapest_price", None):
            return None

        return {
            "price": float(result.cheapest_price),
            "airline": getattr(result, "airline", "SERP") or "SERP",
            "source": "serpapi",
        }


    def _persist_multi_snapshot(
        db: Session,
        group: RouteGroup,
        legs: list[RouteGroupLeg],
        best_combo: tuple[date, ...],
        prices: list[dict[str, Any]],
    ) -> FlightSnapshot:
        total = sum(p["price"] for p in prices)
        details = {
            "total_price": total,
            "legs": [
                {
                    "order": leg.order,
                    "origin": leg.origin,
                    "destination": leg.destination,
                    "date": best_combo[i].isoformat(),
                    "price": prices[i]["price"],
                    "airline": prices[i]["airline"],
                }
                for i, leg in enumerate(legs)
            ],
        }
        snapshot = FlightSnapshot(
            route_group_id=group.id,
            origin=legs[0].origin,
            destination=legs[-1].destination,
            departure_date=best_combo[0],
            return_date=best_combo[-1],
            price=total,
            currency="BRL",
            airline="MULTI",
            source="multi_leg",
            details=details,
        )
        # Dedup via snapshot_service (ver Task 3 branch airline=MULTI)
        if snapshot_service.is_duplicate_snapshot(
            db,
            route_group_id=group.id,
            origin=legs[0].origin,
            destination=legs[-1].destination,
            departure_date=best_combo[0],
            return_date=best_combo[-1],
            price=total,
            airline="MULTI",
        ):
            logger.info("multi_leg: snapshot duplicado ignorado para group %s", group.id)
            return snapshot  # retorna objeto mas nao persiste duplicata
        db.add(snapshot)
        db.commit()
        db.refresh(snapshot)
        return snapshot


    def search_multi_leg_prices(db: Session, group: RouteGroup) -> FlightSnapshot | None:
        legs = sorted(group.legs, key=lambda l: l.order)
        if len(legs) < 2:
            logger.warning("multi_leg: group %s tem menos de 2 legs, ignorando", group.id)
            return None

        candidate_sets = [_candidate_dates(leg) for leg in legs]
        if any(len(c) == 0 for c in candidate_sets):
            return None

        best_total: float | None = None
        best_combo: tuple[date, ...] | None = None
        best_prices: list[dict[str, Any]] | None = None

        passengers = group.passengers or 1

        for combo in itertools.product(*candidate_sets):
            if not _is_valid_chain(combo, legs):
                continue
            leg_prices: list[dict[str, Any]] = []
            for leg, dep_date in zip(legs, combo):
                p = _fetch_leg_price(db, leg, dep_date, passengers)
                if p is None:
                    leg_prices = []
                    break
                leg_prices.append(p)
            if not leg_prices or len(leg_prices) != len(legs):
                continue
            total = sum(p["price"] for p in leg_prices)
            if best_total is None or total < best_total:
                best_total = total
                best_combo = combo
                best_prices = leg_prices

        if best_combo is None or best_prices is None:
            logger.info("multi_leg: nenhuma combinacao valida para group %s", group.id)
            return None

        return _persist_multi_snapshot(db, group, legs, best_combo, best_prices)
    ```
