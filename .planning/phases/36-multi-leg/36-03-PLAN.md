    6. Adicionar `<script>` inline vanilla no final do form — este script sera REUSADO em edit.html (Task 2b):
    ```javascript
    (function() {
      const MAX_LEGS = 5;
      const MIN_LEGS = 2;
      const container = document.getElementById('legs-container');
      const addBtn = document.getElementById('add-leg-btn');
      const template = document.getElementById('leg-template');
      const modeInput = document.getElementById('mode-input');
      const submitBtn = document.getElementById('submit-btn');
      const sectionRound = document.querySelector('section[data-mode="roundtrip"]');
      const sectionMulti = document.querySelector('section[data-mode="multi"]');

      function currentLegs() { return container.querySelectorAll('fieldset.leg'); }

      function addLeg() {
        const legs = currentLegs();
        if (legs.length >= MAX_LEGS) return;
        const idx = legs.length;
        const num = idx + 1;
        const html = template.innerHTML
          .replace(/__N__/g, String(num))
          .replace(/__I__/g, String(idx));
        const wrapper = document.createElement('div');
        wrapper.innerHTML = html;
        container.appendChild(wrapper.firstElementChild);
        renumberAndToggle();
        recalcLegs();
      }

      function removeLeg(fieldset) {
        if (currentLegs().length <= MIN_LEGS) return;
        fieldset.remove();
        renumberAndToggle();
        recalcLegs();
      }

      function renumberAndToggle() {
        const legs = currentLegs();
        legs.forEach((fs, i) => {
          fs.dataset.order = i + 1;
          fs.querySelector('.leg-num').textContent = i + 1;
          fs.querySelectorAll('[name^="legs["]').forEach(inp => {
            inp.name = inp.name.replace(/legs\[\d+\]/, `legs[${i}]`);
          });
          const removeBtn = fs.querySelector('.remove-leg');
          removeBtn.style.display = legs.length > MIN_LEGS ? '' : 'none';
          removeBtn.setAttribute('aria-label', `Remover trecho ${i + 1}`);
        });
        addBtn.style.display = legs.length >= MAX_LEGS ? 'none' : '';
        addBtn.setAttribute('aria-label', `Adicionar trecho ${legs.length + 1}`);
      }

      function parseDate(s) { return s ? new Date(s + 'T00:00:00') : null; }
      function fmtDate(d) {
        const y = d.getFullYear(); const m = String(d.getMonth()+1).padStart(2,'0'); const dd = String(d.getDate()).padStart(2,'0');
        return `${y}-${m}-${dd}`;
      }
      function fmtDateBR(d) {
        const dd = String(d.getDate()).padStart(2,'0'); const m = String(d.getMonth()+1).padStart(2,'0'); const y = d.getFullYear();
        return `${dd}/${m}/${y}`;
      }

      function recalcLegs() {
        const legs = Array.from(currentLegs());
        for (let i = 1; i < legs.length; i++) {
          const fs = legs[i];
          if (fs.dataset.manualOverride === 'true') continue;
          const prev = legs[i-1];
          const prevEnd = parseDate(prev.querySelector('[data-field="window_end"]').value);
          const prevMin = parseInt(prev.querySelector('[data-field="min_stay_days"]').value, 10) || 1;
          const prevMaxRaw = prev.querySelector('[data-field="max_stay_days"]').value;
          const prevMax = prevMaxRaw ? parseInt(prevMaxRaw, 10) : 30;
          if (!prevEnd) continue;
          const wStart = new Date(prevEnd.getTime() + prevMin * 86400000);
          const wEnd = new Date(wStart.getTime() + prevMax * 86400000);
          fs.querySelector('[data-field="window_start"]').value = fmtDate(wStart);
          fs.querySelector('[data-field="window_end"]').value = fmtDate(wEnd);
        }
        validateChainLocal();
      }

      function validateChainLocal() {
        const legs = Array.from(currentLegs());
        for (let i = 1; i < legs.length; i++) {
          const fs = legs[i];
          const prev = legs[i-1];
          const err = fs.querySelector('.leg-error');
          err.textContent = '';
          const prevEnd = parseDate(prev.querySelector('[data-field="window_end"]').value);
          const prevMin = parseInt(prev.querySelector('[data-field="min_stay_days"]').value, 10) || 1;
          const curStart = parseDate(fs.querySelector('[data-field="window_start"]').value);
          if (!prevEnd || !curStart) continue;
          const minStart = new Date(prevEnd.getTime() + prevMin * 86400000);
          if (curStart < minStart) {
            err.textContent = `Trecho ${i+1} precisa sair em ou apos ${fmtDateBR(minStart)}. Ajuste a janela ou reduza a estadia minima do trecho ${i}.`;
          }
        }
      }

      function setMode(mode) {
        modeInput.value = mode;
        if (mode === 'multi_leg') {
          sectionRound.style.display = 'none';
          sectionMulti.style.display = '';
          if (currentLegs().length === 0) { addLeg(); addLeg(); }
          submitBtn.textContent = 'Criar grupo multi-trecho';
        } else {
          sectionRound.style.display = '';
          sectionMulti.style.display = 'none';
          submitBtn.textContent = 'Criar grupo';
        }
        document.querySelectorAll('.mode-btn').forEach(b => {
          b.classList.toggle('active', b.dataset.mode === mode);
        });
      }

      document.querySelectorAll('.mode-btn').forEach(b => {
        b.addEventListener('click', () => setMode(b.dataset.mode));
      });
      addBtn.addEventListener('click', addLeg);
      container.addEventListener('click', e => {
        if (e.target.classList.contains('remove-leg')) removeLeg(e.target.closest('fieldset.leg'));
      });
      container.addEventListener('input', e => {
        const fs = e.target.closest('fieldset.leg');
        if (!fs) return;
        if (e.target.matches('[data-field="window_start"], [data-field="window_end"]')) {
          const legs = Array.from(currentLegs());
          const idx = legs.indexOf(fs);
          if (idx > 0) fs.dataset.manualOverride = 'true';
        }
        if (e.target.matches('[data-field="window_end"], [data-field="min_stay_days"], [data-field="max_stay_days"]')) {
          recalcLegs();
        }
        validateChainLocal();
      });

      // Preload hook para edit.html (Task 2b): se window.__PRELOADED_LEGS__ existir, popular e forcar modo multi
      if (window.__PRELOADED_LEGS__) {
        setMode('multi_leg');
        container.innerHTML = '';
        window.__PRELOADED_LEGS__.forEach((leg, idx) => {
          addLeg();
          const fs = currentLegs()[idx];
          fs.querySelector('[data-field="origin"]').value = leg.origin;
          fs.querySelector('[data-field="destination"]').value = leg.destination;
          fs.querySelector('[data-field="window_start"]').value = leg.window_start;
          fs.querySelector('[data-field="window_end"]').value = leg.window_end;
          fs.querySelector('[data-field="min_stay_days"]').value = leg.min_stay_days;
          if (leg.max_stay_days !== null) fs.querySelector('[data-field="max_stay_days"]').value = leg.max_stay_days;
          fs.dataset.manualOverride = 'true';
        });
      } else {
        // Inicializar modo baseado no hidden input (preserva pre-selecao)
        setMode(modeInput.value || 'roundtrip');
      }
    })();
    ```
